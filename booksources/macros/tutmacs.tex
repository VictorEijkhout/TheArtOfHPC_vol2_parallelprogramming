\newcommand\practical[3]{%
  \par \medskip
    {\bf Exercise. }#1\par % do this
  \smallskip
  \def\next{#2}\def\empty{}
  \ifx\next\empty\else
    {\sl Expected outcome. }#2\par
  \fi
  \def\next{#3}\def\empty{}
  \ifx\next\empty\else
    \smallskip
    {\sl Caveats. }#3\par
  \fi
  \medskip
}
