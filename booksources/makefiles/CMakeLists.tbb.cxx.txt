cmake_minimum_required( VERSION 3.20 )
project( ${PROJECTNAME} VERSION 1.0 )

add_executable( ${PROJECTNAME} ${PROJECTNAME}.cxx )
target_compile_features( ${PROJECTNAME} PRIVATE cxx_std_17 )

#### cxxopts
find_package( PkgConfig REQUIRED )
pkg_check_modules( CXXOPTS REQUIRED cxxopts )
target_include_directories( ${PROJECTNAME} PUBLIC ${CXXOPTS_INCLUDE_DIRS} )

#### OpenMP linking
find_package(OpenMP REQUIRED)
target_link_libraries( ${PROJECTNAME} PUBLIC OpenMP::OpenMP_CXX )

#### TBB linking
find_package(TBB REQUIRED)
target_link_libraries( ${PROJECTNAME} PUBLIC TBB::tbb)

install( TARGETS ${PROJECTNAME} DESTINATION . )
